# Old Age Home Management System in HTML Based on common features from similar projects on GitHub, I've created a comprehensive static HTML-based Old Age Home Management System. Since it's purely HTML, CSS, and JavaScript (no backend), it uses localStorage to simulate data persistence across pages. This allows for CRUD operations on residents, staff, rooms, medical records, and more. The system includes: - **User Roles**: Admin (manage everything), Doctor/Staff (view and update medical records, appointments), Resident/Guardian (view personal details, request services). - **Features**: - Login simulation (no real auth; selects role and redirects). - Dashboards for each role. - Resident management: Add, view, edit, delete residents. - Staff management: Similar for staff. - Room management: Track occupancy. - Medical records and appointments. - Contact/Feedback form (simulates submission). - Search functionality. - About, Services, Rules, Eligibility pages. - **Technologies**: HTML5, CSS3 (with Bootstrap 5 for responsive design), JavaScript (for interactivity and localStorage). To run: Save the files in a folder and open `index.html` in a browser. Data persists in the browser's localStorage. ## File Structure - `index.html`: Home/Login page. - `admin.html`: Admin dashboard. - `doctor.html`: Doctor/Staff dashboard. - `resident.html`: Resident/Guardian dashboard. - `about.html`, `services.html`, `rules.html`, `eligibility.html`, `contact.html`: Static info pages. - `styles.css`: Custom CSS (in addition to Bootstrap). - `script.js`: Shared JavaScript for all pages. Redirecting to home.'); window.location.href = 'index.html'; } } function logout() { localStorage.removeItem('role'); window.location.href = 'index.html'; } // Data storage keys const keys = { residents: 'residents', staff: 'staff', rooms: 'rooms', appointments: 'appointments', feedback: 'feedback' }; // Load data function loadData(key) { return JSON.parse(localStorage.getItem(key)) || []; } function saveData(key, data) { localStorage.setItem(key, JSON.stringify(data)); } // General CRUD functions function addItem(key, item) { const items = loadData(key); item.id = items.length ? items[items.length - 1].id + 1 : 1; items.push(item); saveData(key, items); } function updateItem(key, updatedItem) { const items = loadData(key); const index = items.findIndex(i => i.id == updatedItem.id); if (index !== -1) { items[index] = updatedItem; saveData(key, items); } } function deleteItem(key, id) { let items = loadData(key); items = items.filter(i => i.id != id); saveData(key, items); } // Load all tables function loadAllData() { if (document.getElementById('residentsTable')) loadResidents(); if (document.getElementById('staffTable')) loadStaff(); if (document.getElementById('roomsTable')) loadRooms(); if (document.getElementById('appointmentsTable')) loadAppointments(); if (document.getElementById('feedbackTable')) loadFeedback(); const table = document.getElementById('residentsTable'); table.innerHTML = ''; residents.forEach(r => { const row = ` ${r.id} ${r.name} ${r.age} ${r.contact} ${r.medical} `; table.innerHTML += row; }); } function editResident(id) { const residents = loadData(keys.residents); const r = residents.find(r => r.id == id); document.getElementById('residentId').value = r.id; document.getElementById('residentName').value = r.name; } else { addItem(keys.staff, staff); } clearForm('staffForm'); loadStaff(); } function loadStaff() { // Similar to loadResidents const staff = loadData(keys.staff); const table = document.getElementById('staffTable'); table.innerHTML = ''; staff.forEach(s => { const row = ` ${s.id} ${s.name} ${s.role} ${s.contact} `; table.innerHTML += row; }); } function editStaff(id) { // Similar } function saveRoom() { const id = document.getElementById('roomId').value; updateItem(keys.rooms, room); } else { addItem(keys.rooms, room); } clearForm('roomForm'); loadRooms(); } function loadRooms() { const rooms = loadData(keys.rooms); const table = document.getElementById('roomsTable'); table.innerHTML = ''; rooms.forEach(r => { const row = ` ${r.id} ${r.number} ${r.status} ${r.resident || 'None'} `; table.innerHTML += row; }); } function editRoom(id) { // Similar } function saveAppointment() { const id = document.getElementById('appointmentId').value; } function editAppointment(id) { // Similar } function saveFeedback() { const feedback = { name: document.getElementById('contactName').value, email: document.getElementById('contactEmail').value, message: document.getElementById('contactMessage').value }; addItem(keys.feedback, feedback); alert('Feedback submitted!'); clearForm('contactForm'); } function loadFeedback() { const feedbacks = loadData(keys.feedback); const table = document.getElementById('feedbackTable'); table.innerHTML = ''; feedbacks.forEach(f => { const row = ` ${f.id} ${f.name} ${f.email} ${f.message} `; table.innerHTML += row; }); } // For doctor page function loadDocAppointments() { // Assume doctor ID is 1 for simulation; saveData(keys.residents, residents); alert('Medical history updated!'); } else { alert('Resident not found.'); } } // For resident page function viewMyDetails() { const resId = document.getElementById('resId').value;
